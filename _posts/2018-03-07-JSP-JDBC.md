# 방법1

## 1단계. Java Resources> src> 패키지 생성> 연결 클래스 생성

```java
package com.test;

import java.sql.*;

//데이터베이스 연결 정보 관리 클래스
public class MySQLConnection {

	// JDBC driver name and database URL
	static final String JDBC_DRIVER = "com.mysql.jdbc.Driver";
	static final String DB_URL = "jdbc:mysql://211.63.89.72:3306/test03?useSSL=false"; // 연결문자열(포트번호까지 작성)

	// Database credentials
	static final String USER = "test03";
	static final String PASS = "1234";

	private static Connection conn;

	// 데이터베이스 커넥션 객체 생성 메소드
	public static Connection connect() throws ClassNotFoundException, SQLException {

		// STEP 2: Register JDBC driver
		Class.forName(JDBC_DRIVER);
		// STEP 3: Open a connection
		conn = DriverManager.getConnection(DB_URL, USER, PASS);

		return conn;
	}

	// 데이테베이스 커넥션 객체 소멸 메소드
	public static void close() throws SQLException {
		if (conn != null) {
			conn.close();
		}
	}

}
```

## 2단계. 1단계와 같은 위치에 DAO클래스 생성

```java
package com.test;

import java.sql.*;
import java.util.*;

public class MemberDAO {

	// 출력 액션 메소드
	public void method() {

		Connection conn = null;
		PreparedStatement pstmt = null;

		try {
			conn = MySQLConnection.connect();
			System.out.println("연결 성공");
		} catch (SQLException se) {
			se.printStackTrace();
		} catch (Exception e) {
			e.printStackTrace();
		} finally {
			try {
				if (pstmt != null)
					pstmt.close();
			} catch (SQLException se) {
			}
			try {
				MySQLConnection.close();
			} catch (SQLException se) {
				System.out.println(se.getMessage());
			}
		}
	}
}

```
