```python
a= [1,2,3]

# append는 맨 마지막에 값 삽입
a.append(4)
print(a)

# insert(b, c)은 b번째 위치에 c를 삽입하는 함수
a.insert(0, 0)
print(a)

# remove(요소값)은 a 리스트에서 특정 요소값과 일치하는 첫번째 인덱스를 삭제한다.
a.remove(3)
print(a)

# del은 요소 삭제 함수
del(a[1])
print(a)

# pop은 마지막 요소를 반환하고 제거(꺼내기)
# 중간 값도 인덱스를 지정해 꺼낼 수 있음 (ex. a.pop(1))
b=a.pop()
print(a)    # 남은 것
print(b)    # 꺼내진 값
```

```python
#결과
[1, 2, 3, 4]
[0, 1, 2, 3, 4]
[0, 1, 2, 4]
[0, 2, 4]
[0, 2]
4
```



# 딕셔너리 자료형

```python
a = {}  # a = dict()와 같다 (빈 딕셔너리 만드는 방법은 두 가지)
c = list()
print(a)

# 주의) 대소문자 구분한다
# 주의) key값으로 list 자료형은 올 수 없다
a['name'] = 'hong'
print(a)

a['phone'] = '010-1234-1234'
print(a)

print(a['name'])
print(a['phone'])


a['name'] = 'park'
print(a)

# key값/ value값만 뽑아내기
print(a.keys())
print(a.values())

# list형으로 형 변환
print(list(a.keys()))
print(list(a.values()))

b =list(a.values())
print(b[0])

# 튜플로 변환...key-value 기능은 없어져
print(a.items())

# a[없는키]와 a.get(없는키)는 다르다
# print(a['nokey']) -> 오류
print(a.get('nokey')) # -> none
print(a.get('nokey','검색결과없음')) # -> none일 경우 대체할 값 설정


```

```python
#결과
{}
{'name': 'hong'}
{'name': 'hong', 'phone': '010-1234-1234'}
hong
010-1234-1234
{'name': 'park', 'phone': '010-1234-1234'}
dict_keys(['name', 'phone'])
dict_values(['park', '010-1234-1234'])
['name', 'phone']
['park', '010-1234-1234']
park
dict_items([('name', 'park'), ('phone', '010-1234-1234')])
None
검색결과없음
```
